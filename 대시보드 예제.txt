                const m = String(date.getMonth() + 1).padStart(2, '0');
                const d = String(date.getDate()).padStart(2, '0');
                return `${m}/${d}`;
              }

              function compressDates(dates) {
                const uniqueFormatted = Array.from(new Set(
                  dates.map(d => parseDate(d)).filter(Boolean).map(d => formatDate(d))
                ));

                const parsed = uniqueFormatted
                  .map(dStr => new Date(`${new Date().getFullYear()}-${dStr.replace('/', '-')}`))
                  .sort((a, b) => a - b);

                if (!parsed.length) return dates;

                const result = [];
                let start = parsed[0];
                let prev = parsed[0];

                for (let i = 1; i < parsed.length; i++) {
                  const curr = parsed[i];
                  const diffDays = (curr - prev) / (1000 * 60 * 60 * 24);

                  if (diffDays === 1) {
                    prev = curr;
                  } else {
                    const duration = Math.round((prev - start) / (1000 * 60 * 60 * 24)) + 1;
                    result.push(`${formatDate(start)}~${formatDate(prev)} <span style="font-size:11px; color:#FFD54F; font-weight:600;">(${duration}일)</span>`);
                    start = curr;
                    prev = curr;
                  }
                }

                const duration = Math.round((prev - start) / (1000 * 60 * 60 * 24)) + 1;
                if (formatDate(start) === formatDate(prev)) {
                  result.push(`${formatDate(start)} <span style="font-size:11px; color:#FFD54F; font-weight:600;">(1일)</span>`);
                } else {
                  result.push(`${formatDate(start)}~${formatDate(prev)} <span style="font-size:11px; color:#FFD54F; font-weight:600;">(${duration}일)</span>`);
                }

                return result;
